function countdown(n){ 
  if (n < 1) {
    return [];
  } else {
    const countArray = countdown(n-1);
    console.log(countArray)
    countArray.unshift(n);
    return countArray;
  }
}
// Only change code above this line
console.log(countdown(5)); // [5, 4, 3, 2, 1]
